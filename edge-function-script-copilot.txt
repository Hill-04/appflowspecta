serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response(null, { headers: corsHeaders });
  }

  try {
    // ========== AUTH VALIDATION ==========
    const authHeader = req.headers.get("Authorization");
    if (!authHeader?.startsWith("Bearer ")) {
      return new Response(JSON.stringify({ error: "Missing authorization" }), {
        status: 401,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    const supabaseClient = createClient(
      Deno.env.get("SUPABASE_URL") ?? "",
      Deno.env.get("SUPABASE_ANON_KEY") ?? "",
      { global: { headers: { Authorization: authHeader } } }
    );

    const token = authHeader.replace("Bearer ", "");
    const { data: claimsData, error: claimsError } = await supabaseClient.auth.getClaims(token);
    if (claimsError || !claimsData?.claims) {
      return new Response(JSON.stringify({ error: "Unauthorized" }), {
        status: 401,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    // ========== INPUT VALIDATION ==========
    const messageSchema = z.object({
      role: z.enum(["user", "assistant"]),
      content: z.string().max(5000),
    });

    const requestSchema = z.object({
      action: z.enum(["chat", "analyze_script", "generate_block", "refine_block", "generate_variation"]),
      messages: z.array(messageSchema).max(50).optional(),
      context: z.object({
        objective: z.string().max(100).optional(),
        channel: z.string().max(50).optional(),
        existingScript: z.string().max(10000).optional(),
        profileData: z.object({
          offerType: z.string().max(200).optional(),
          targetAudienceDescription: z.string().max(500).optional(),
          mainPain: z.string().max(200).optional(),
          differential: z.string().max(500).optional(),
          averageTicket: z.string().max(100).optional(),
          maturityLevel: z.string().max(100).optional(),
        }).optional(),
      }).optional(),
      block: z.string().max(50).optional(),
      objective: z.string().max(100).optional(),
      channel: z.string().max(50).optional(),
      profileData: z.object({
        offerType: z.string().max(200).optional(),
        targetAudienceDescription: z.string().max(500).optional(),
        mainPain: z.string().max(200).optional(),
        differential: z.string().max(500).optional(),
        averageTicket: z.string().max(100).optional(),
      }).optional(),
      feedback: z.string().max(2000).optional(),
      currentContent: z.string().max(5000).optional(),
    });

    let body: z.infer<typeof requestSchema>;
    try {
      body = requestSchema.parse(await req.json());
    } catch (e) {
      return new Response(JSON.stringify({ error: "Invalid request format" }), {
        status: 400,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    const { action } = body;

    const LOVABLE_API_KEY = Deno.env.get("LOVABLE_API_KEY");
    if (!LOVABLE_API_KEY) {
      return new Response(JSON.stringify({ error: "API key not configured" }), {
        status: 500,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    // ========== CHAT (streaming SSE) ==========
    if (action === "chat") {
      const { messages, context } = body;
      const { objective, channel, profileData } = context || {};

      const profileContext = profileData
        ? `\nPerfil Estrategico do usuario:\n- Tipo de oferta: ${profileData.offerType || "nao informado"}\n- Publico-alvo: ${profileData.targetAudienceDescription || "nao informado"}\n- Dor principal: ${profileData.mainPain || "nao informado"}\n- Diferencial: ${profileData.differential || "nao informado"}\n- Ticket medio: ${profileData.averageTicket || "nao informado"}\n- Canal: ${channel || "nao informado"}\n- Maturidade: ${profileData.maturityLevel || "nao informado"}`
        : "";

      const objectiveInstr = objective ? (OBJECTIVE_INSTRUCTIONS[objective] || `OBJETIVO: ${objective}`) : "";
      const channelInstr = channel ? (CHANNEL_PLACEHOLDER_INSTRUCTIONS[channel] || "") : "";

      const systemPrompt = `${CHAT_SYSTEM_PROMPT}\n\n${profileContext}\n\n${objectiveInstr}\n\n${channelInstr}`.trim();

      const response = await fetch("https://ai.gateway.lovable.dev/v1/chat/completions", {
        method: "POST",
        headers: {
          Authorization: `Bearer ${LOVABLE_API_KEY}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          model: "google/gemini-3-flash-preview",
          messages: [
            { role: "system", content: systemPrompt },
            ...(messages || []),
          ],
          stream: true,
          max_tokens: 4096,
        }),
      });

      if (!response.ok) {
        const status = response.status;
        if (status === 429) {
          return new Response(JSON.stringify({ error: "Limite de requisiÃ§Ãµes atingido. Tente novamente em alguns segundos." }), {
            status: 429,
            headers: { ...corsHeaders, "Content-Type": "application/json" },
          });
        }
        if (status === 402) {
          return new Response(JSON.stringify({ error: "CrÃ©ditos insuficientes. Adicione crÃ©ditos para continuar." }), {
            status: 402,
            headers: { ...corsHeaders, "Content-Type": "application/json" },
          });
        }
        const t = await response.text();
        console.error("AI gateway error:", status, t);
        return new Response(JSON.stringify({ error: "Erro ao gerar conteÃºdo" }), {
          status: 500,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        });
      }

      return new Response(response.body, {
        headers: { ...corsHeaders, "Content-Type": "text/event-stream" },
      });
    }

    // ========== ANALYZE SCRIPT (non-streaming) ==========
    if (action === "analyze_script") {
      const { context } = body;
      const { existingScript, objective, channel, profileData } = context || {};

      if (!existingScript || existingScript.trim().length < 20) {
        return new Response(JSON.stringify({ content: "O script estÃ¡ muito curto para anÃ¡lise. Cole um script mais completo para que eu possa avaliar e sugerir melhorias." }), {
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        });
      }

      const truncated = existingScript.slice(0, 5000);

      const profileContext = profileData
        ? `\nPerfil do usuario:\n- Oferta: ${profileData.offerType || "?"}\n- Publico: ${profileData.targetAudienceDescription || "?"}\n- Dor: ${profileData.mainPain || "?"}\n- Diferencial: ${profileData.differential || "?"}\n- Canal: ${channel || "?"}\n- Objetivo: ${objective || "?"}`
        : "";

      const userPrompt = `Analise este script de prospecÃ§Ã£o e sugira melhorias:\n\n${truncated}`;

      const response = await fetch("https://ai.gateway.lovable.dev/v1/chat/completions", {
        method: "POST",
        headers: {
          Authorization: `Bearer ${LOVABLE_API_KEY}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          model: "google/gemini-3-flash-preview",
          messages: [
            { role: "system", content: `${ANALYZE_SYSTEM_PROMPT}\n${profileContext}` },
            { role: "user", content: userPrompt },
          ],
          max_tokens: 4096,
        }),
      });

      if (!response.ok) {
        const status = response.status;
        if (status === 429) {
          return new Response(JSON.stringify({ error: "Limite de requisiÃ§Ãµes atingido." }), { status: 429, headers: { ...corsHeaders, "Content-Type": "application/json" } });
        }
        if (status === 402) {
          return new Response(JSON.stringify({ error: "CrÃ©ditos insuficientes." }), { status: 402, headers: { ...corsHeaders, "Content-Type": "application/json" } });
        }
        const t = await response.text();
        console.error("AI gateway error:", status, t);
        return new Response(JSON.stringify({ error: "Erro ao analisar script" }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } });
      }

      const data = await response.json();
      const content = data.choices?.[0]?.message?.content || "";
      return new Response(JSON.stringify({ content }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    // ========== LEGACY BLOCK ACTIONS (generate_block, refine_block, generate_variation) ==========
    const { block, objective, channel, profileData, currentContent, feedback } = body;

    const profileContext = profileData
      ? `\nPerfil Estrategico do usuario:\n- Tipo de oferta: ${profileData.offerType || "nao informado"}\n- Publico-alvo: ${profileData.targetAudienceDescription || "nao informado"}\n- Dor principal do publico: ${profileData.mainPain || "nao informado"}\n- Diferencial: ${profileData.differential || "nao informado"}\n- Ticket medio: ${profileData.averageTicket || "nao informado"}\n- Canal de contato: ${channel || "nao informado"}`
      : "";

    const objectiveInstr = OBJECTIVE_INSTRUCTIONS[objective] || "";
    const blockInstr = BLOCK_INSTRUCTIONS[block] || "";

    let userPrompt = "";

    if (action === "generate_block") {
      userPrompt = `${blockInstr}\n\n${objectiveInstr}\n\nGere o bloco agora. Use os placeholders [NomeLead], [Empresa], [DorPrincipal], [Diferencial], [CTA] quando apropriado. Maximo 3 paragrafos curtos. Linguagem natural, conversacional, sem parecer robo.`;
    } else if (action === "refine_block") {
      userPrompt = `Bloco atual:\n"${currentContent}"\n\nFeedback do usuario: "${feedback}"\n\n${objectiveInstr}\n\nRefine o bloco acima aplicando o feedback. Mantenha os placeholders existentes. Maximo 3 paragrafos.`;
    } else if (action === "generate_variation") {
      userPrompt = `Bloco atual:\n"${currentContent}"\n\n${objectiveInstr}\n\nGere uma VARIACAO ALTERNATIVA deste bloco com abordagem diferente mas mesmo objetivo. Use placeholders quando apropriado. Maximo 3 paragrafos.`;
    } else {
      return new Response(JSON.stringify({ error: "Invalid action" }), {
        status: 400,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    const systemPrompt = `Voce e um especialista em copywriting para prospeccao B2B e vendas consultivas. Voce cria mensagens de prospeccao que parecem humanas, diretas e personalizadas. Nunca use jargoes de marketing generico. Sempre escreva em portugues brasileiro.\n\n${profileContext}`;

    const response = await fetch("https://ai.gateway.lovable.dev/v1/chat/completions", {
      method: "POST",
      headers: {
        Authorization: `Bearer ${LOVABLE_API_KEY}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        model: "google/gemini-3-flash-preview",
        messages: [
          { role: "system", content: systemPrompt },
          { role: "user", content: userPrompt },
        ],
        tools: [
          {
            type: "function",
            function: {
              name: "return_block",
              description: "Return the generated script block content",
              parameters: {
                type: "object",
                properties: {
                  content: { type: "string", description: "The generated block text" },
                },
                required: ["content"],
                additionalProperties: false,
              },
            },
          },
        ],
        tool_choice: { type: "function", function: { name: "return_block" } },
      }),
    });

    if (!response.ok) {
      const status = response.status;
      if (status === 429) {
        return new Response(JSON.stringify({ error: "Limite de requisiÃ§Ãµes atingido. Tente novamente em alguns segundos." }), {
          status: 429,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        });
      }
      if (status === 402) {
        return new Response(JSON.stringify({ error: "CrÃ©ditos insuficientes. Adicione crÃ©ditos para continuar." }), {
          status: 402,
          headers: { ...corsHeaders, "Content-Type": "application/json" },
        });
      }
      const t = await response.text();
      console.error("AI gateway error:", status, t);
      return new Response(JSON.stringify({ error: "Erro ao gerar conteÃºdo" }), {
        status: 500,
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    const data = await response.json();
    const toolCall = data.choices?.[0]?.message?.tool_calls?.[0];

    if (toolCall?.function?.arguments) {
      const args = JSON.parse(toolCall.function.arguments);
      return new Response(JSON.stringify({ content: args.content }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    const content = data.choices?.[0]?.message?.content || "";
    return new Response(JSON.stringify({ content }), {
      headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  } catch (e) {
    console.error("script-copilot error:", e);
    return new Response(JSON.stringify({ error: e instanceof Error ? e.message : "Unknown error" }), {
      status: 500,
      headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  }
});
